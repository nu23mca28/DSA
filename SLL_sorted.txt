#include<stdio.h>
struct node{
    int info;
    struct node *next;
};

typedef struct node *nodeptr;
nodeptr p,list=NULL,temp;

nodeptr getnode();
void insert_b();
void delete_b();
void display();

void main(){
    int c;
    char ch;
    do{
        printf("\n\n1.Insert beginning\n2.Delete beginning\n3.Display\n4.Exit\n");
        printf("Enter your choice : ");
        scanf("%d",&c);
        switch(c){
            case 1:insert_b();
                    break;
            case 2:delete_b();
                    break;
            case 3:display();
                    break;
            case 4:printf("Thankyou!");
                    break;
            default:printf("Invalid choice");
        }
    }while(c!=4);
}

nodeptr getnode(){
    return ((struct node*)malloc(sizeof(struct node)));
}

void insert_b(){
    int x;
    nodeptr p,q;
    printf("Enter the element : ");
    scanf("%d",&x);
    p=list;
    temp=getnode();
    temp->info=x;
    if(p==NULL){
        temp->next=p;
        list=temp;
        return;
    }
    if(p->next==NULL){
        if(x <= p->info){
            temp->next=p;
            list=temp;
            return;
        }
        else{
            temp->next=NULL;
            p->next=temp;
        }
    }
    while(p!=NULL){
        if(p->next!=NULL && p->info >= x){
            break;
        }
        q=p;
        p=p->next;
    }
    q->next=temp;
    temp->next=p;
    if(p==NULL){
        return;
    }
}

void delete_b(){
    p=list;
    if(list==NULL){
        printf("No elements");
        return;
    }
    printf("%d Deleted from the list",p->info);
    list=p->next;
    p->next=NULL;
    free(p);
}

void display(){
    if(list==NULL){
        printf("No elemenets");
    }
    else{
        p=list;
        printf("List elements are : ");
        while(p!=NULL){
            printf("%d ",p->info);
            p=p->next;
        }
    }
}
