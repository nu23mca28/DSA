#include<stdio.h>
struct node{
    char name[50];
    int reg;
    struct node *next;
};

typedef struct node *nodeptr;
nodeptr p,list=NULL,temp;

nodeptr getnode();
void insert_b();
void delete_b();
void display();

void main(){
    int c;
    char ch;
    do{
        printf("\n\n1.Push\n2.Pop\n3.Display\n4.Exit\n");
        printf("Enter your choice : ");
        scanf("%d",&c);
        switch(c){
            case 1:insert_b();
                    break;
            case 2:delete_b();
                    break;
            case 3:display();
                    break;
            case 4:printf("Thankyou!");
                    break;
            default:printf("Invalid choice");
        }
    }while(c!=4);
}

nodeptr getnode(){
    return ((struct node*)malloc(sizeof(struct node)));
}

void insert_b(){
    int r;
    char n[50];
    printf("Enter the Student RegNo and Name : ");
    scanf("%d",&r);
    scanf("%s",n);
    temp=getnode();
    temp->reg=r;
    strcpy(temp->name,n);
    temp->next=list;
    list=temp;
}

void delete_b(){
    p=list;
    if(p==NULL){
        printf("No students in the list");
        return;
    }
    printf("%d %s removed from List",p->reg,p->name);
    list=p->next;
    p->next=NULL;
    free(p);
}

void display(){
    if(list==NULL){
        printf("No students in the list");
    }
    else{
        p=list;
        printf("RegNo\t Name\n");
        while(p!=NULL){
            printf("%d\t %s\n",p->reg,p->name);
            p=p->next;
        }
    }
}
